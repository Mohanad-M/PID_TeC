41void MW_Arduino_Init(void)



42{



43    //Disable watchdog for Arduino Due. watchdog 



44    #if (defined(_ROTH_DUE_))



45      watchdogDisable();



46    #endif



47  



48    /* 



49     * Initialize Serial Peripheral



50     */



51    // Always Initialize Serial0 



52    Serial_begin(0, _RTT_BAUDRATE_SERIAL0_);



53    



54



55    #ifdef _RTT_USE_SERIAL1_



56        Serial_begin(1, _RTT_BAUDRATE_SERIAL1_);



57    #endif



58        



59    #ifdef _RTT_USE_SERIAL2_



60        Serial_begin(2, _RTT_BAUDRATE_SERIAL2_);



61    #endif



62



63    #ifdef _RTT_USE_SERIAL3_



64        Serial_begin(3, _RTT_BAUDRATE_SERIAL3_);



65    #endif



66        



67    /* Overrun Detection Initialization */    



68    #ifdef _RTT_OVERRUN_DIGITAL_PIN_



69        pinMode(_RTT_OVERRUN_DIGITAL_PIN_, OUTPUT);



70        



71        /* Init the overrun detection pin to zero or else the Pin is always in a



72           high/tristate giving the user a wrong indication of overrun conditions */



73        digitalWrite(_RTT_OVERRUN_DIGITAL_PIN_, LOW);



74    #endif



75



76    /*



77     * Initialize Analog Reference Volatge



78     * Default - Due



79     * Default, Internal 1.1, Internal 2.56, External -  Mega 2560, Mega ADK



80     * Deafult, Internal, External - Other Arduino targets



81     */    



82    #if (defined(_ROTH_MEGA2560_))



83        #if (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 1))



84             analogReference(INTERNAL1V1);



85        #elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 2))



86             analogReference(INTERNAL2V56);



87        #elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 3))



88             analogReference(EXTERNAL);



89        #endif



90    #elif defined(_ROTH_MKR1000_) || defined(_ROTH_MKRWIFI1010_) || defined(_ROTH_MKRZERO_) || defined(_ROTH_NANO33_IOT_)



91        #if (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 1))



92             analogReference(AR_INTERNAL1V0);



93        #elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 2))



94             analogReference(AR_INTERNAL1V65);



95        #elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 3))



96             analogReference(AR_INTERNAL2V23);



97        #elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 4))



98             analogReference(AR_EXTERNAL);



99        #endif



100    #else 



101        #if (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 1))



102            analogReference(INTERNAL);



103        #elif (defined(_RTT_ANALOG_REF_) && (_RTT_ANALOG_REF_ == 2))



104            analogReference(EXTERNAL);



105        #endif



106    #endif



107             



108    /* 



109     * Initialize SPI bus



110     */



111        //_RTT_SPI_SSPIN_ definition is applicable for all boards excluding Due



112        #if defined(_RTT_SPI_SSPIN_)



113            MW_SPIwriteReadSetup();



114        #endif



115    #ifdef NANO_MOTOR_CARRIER 



116    MW_EnableNanoCarrierBatteryCharging();



117    #endif



118          



119           



120}



121/* Function: MW_Arduino_Terminate ================================================



122 * For Arduino MKR/Leonardo and its variants, the Virtual COM port is handled 



123 * by the controller. In case the code running on the target exits main,



124 * the COM port cannot be accessed until a hard reset is performed.



125 * To over come this issue, a while loop is added to make sure that 



126 * upon getting a stop command from external mode, the code running on



127 * the target stops but the code will not exit the main.



128 * This will ensure that the COM port is accessible even after the 



129 * external mode has been stopped. */



130void MW_Arduino_Terminate(void){



131	#if defined(_ROTH_MKR1000_) || defined(_ROTH_MKRWIFI1010_) || defined(_ROTH_MKRZERO_) || defined(_ROTH_LEONARDO_) || defined(_ROTH_NANO33_IOT_)



132        disable_rt_OneStep();



133		while(1){};     /* To continue virtual COM port functionality */



134	#endif



135}

137#ifdef NANO_MOTOR_CARRIER



138void MW_EnableNanoCarrierBatteryCharging(void) {



139  Wire.begin();



140  Wire.beginTransmission(PMIC_ADDRESS);



141  Wire.write(PMIC_REG00);



142  Wire.write(0x06); // min sys voltage 3.88V + max input current 2.0 A



143  Wire.endTransmission();



144  Wire.beginTransmission(PMIC_ADDRESS);



145  Wire.write(PMIC_REG01);



146  Wire.write(0x1B); // Charge Battery + Minimum System Voltage 3.5V



147  Wire.endTransmission();



148  Wire.beginTransmission(PMIC_ADDRESS);



149  Wire.write(PMIC_REG02);



150  Wire.write(0x00); // Charge current  512 mA



151  Wire.endTransmission();



152  Wire.beginTransmission(PMIC_ADDRESS);



153  Wire.write(PMIC_REG04);



154  Wire.write(0x9E); // Charge Voltage Limit 4.128V



155  Wire.endTransmission();



156  Wire.beginTransmission(PMIC_ADDRESS);



157  Wire.write(PMIC_REG05);



158  Wire.write(0x8A); // Enable Battery Charge termination + disable watchdog



159  Wire.endTransmission();



160}
