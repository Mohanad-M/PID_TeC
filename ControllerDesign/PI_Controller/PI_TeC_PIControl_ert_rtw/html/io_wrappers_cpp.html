137    extern "C" uint8_t MW_EthernetAndUDPBegin(uint8_t udpindex, uint32_t localPort)



138    {       



139        static int only_one_ethernet_begin;



140        if (only_one_ethernet_begin==0)



141        {



142          #if (defined(_RTT_DISABLE_Ethernet_DHCP_) && (_RTT_DISABLE_Ethernet_DHCP_ !=0))



143            Ethernet.begin(mac, localIpAddress);



144            configureSuccess = 1;



145          #else



146            configureSuccess = Ethernet.begin(mac);



147          #endif



148            only_one_ethernet_begin++;



149            



150            for(trialcount=0;trialcount<5;trialcount++)



151            {



152                if(Serial)



153                {



154                    break;



155                }else



156                {



157                    delay(1000);



158                }



159            }



160            // Depending on if the IP address assignment using DHCP has been successful or not,



161            // send the appropriate information back to the host.



162            // If External mode over TCP/IP or Wifi, do not print this Info. Information of IP address will be printed as 



163            // a part of rtiostream_open



164          #if (!(defined(_RTT_TCPIP_EXTMODE_) && (_RTT_TCPIP_EXTMODE_ == 1)))



165                if (configureSuccess==1)



166                {



167                   if(Ethernet.localIP()[0] == 0 || Ethernet.localIP()[0] == 255)



168                    {



169                        /*Tried to assign Static IP, failed due to one of the following reasons :



170                         * 1. Ethernet shield is not connected properly with Arduino.



171                         * 2. Ethernet shield is not connected to a network.



172                         **/



173                        Serial.println("<<<IP address: Failed to configure.>>>");



174                    }



175                    else



176                    {



177                        // If the Configuration is successful, relay back the assigned IP address.



178                        Serial.print("<<<IP address: ");



179                        for (byte thisByte = 0; thisByte < 4; thisByte++)



180                        {



181                            // Print the value of each byte of the IP address:



182                            Serial.print(Ethernet.localIP()[thisByte], DEC);



183                            if (thisByte<3)



184                            {



185                                Serial.print(".");



186                            }



187                        }



188                        Serial.println(">>>");



189                    }     



190                }



191               else



192               {



193                   // If the Configuration failed,relay back the error message.



194                   Serial.println("<<<IP address: Failed to configure.>>>");



195               }



196          #endif



197        }



198        return (Udp[udpindex].begin(localport[udpindex]));



199    }

202    {



203        IPAddress remoteIpAddress(remoteIPOctect1,remoteIPOctect2, remoteIPOctect3, remoteIPOctect4);

359extern "C" void MW_delay_in_milliseconds(uint32_T period)



360{



361    uint32_T start_time = 0;



362    start_time = millis();



363    while(millis()- start_time <= period) 



364    {



365        1;



366    }



367}
